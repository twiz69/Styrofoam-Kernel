#!/usr/bin/env bash

kernel_dir="${PWD}"
CCACHE=$(command -v ccache)
HOME=/home/Android/Styrofoam-Kernel/
objdir="${kernel_dir}/out
builddir="${kernel_dir}/build
ZIMAGE=$kernel_dir/out/arch/arm64/boot/Image.gz-dtb
export ARCH=arm64
export KBUILD_BUILD_USER=Twizz
export KBUILD_BUILD_HOST=Zorin_os
export LD_LIBRARY_PATH="$HOME/android/toolchains/clang/bin/../lib:$PATH"
# Compile plox
function compile() {
    make -s -j$(nproc) O=out merlin_defconfig
    PATH="$HOME/android/toolchains/clang/bin:${PATH}" \
    make -C $(pwd) -j$(nproc) O=out \
                   ARCH=arm64 \
                   CC=clang \
                   CLANG_TRIPLE=aarch64-linux-gnu- \
                   CROSS_COMPILE=aarch64-linux-gnu- \
                   CROSS_COMPILE_ARM32=arm-linux-gnueabi-
}
compile
